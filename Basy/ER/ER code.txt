Table templates {
  id integer [primary key]
  name varchar
  text text
  created_at timestamp
  has_parameters bool
  has_versions bool
}

Table tags {
  id integer [primary key]
  name varchar
}

Table templatesTags {
  template_id integer
  tag_id integer
}

Table parameters {
  id integer [primary key]
  name varchar
  template_id integer
}

Table logs {
  id integer [primary key]
  action text
  timestamp timestamp
}

Table versions {
  id integer [primary key]
  template_id integer
  name varchar
  text text
}
